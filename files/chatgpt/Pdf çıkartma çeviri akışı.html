🧭 Genel Akış Tasarımı

  1. PDF'den yapısal çıkarım (extraction)

    • PDF içeriğini sırayla çıkar (metin, görsel, tablo).

    • Bu akış sırasını koru → [TEXT, IMAGE, TEXT, TABLE, ...]

    • Kullanılacak kütüphaneler:

      • PyMuPDF (fitz) → metin blokları, resimler, sayfa yapısı

      • pdfplumber → tablo tespiti (özellikle grid yapısı için)

      • pandas → tabloyu CSV veya markdown’a çevirmek için.

  2. Akışı JSON formatında sakla

    json
    [
  {"type": "text", "content": "Lorem ipsum..."},
  {"type": "image", "path": "page1_image1.png"},
  {"type": "text", "content": "Nam commodo..."},
  {"type": "table", "content": [["A", "B"], ["1", "2"]]}
]

  3. LLM çeviri isteği gönder

    • Text bloklarını LLM'e gönder (örneğin vLLM veya başka bir serve edilen model endpoint).

    • Her text bloğunun çevirisini al ve JSON içindeki sıraya göre yerleştir.

    • Görsel ve tablo tipleri değişmeden kalır.

    • python-docx ile:

      • text → paragraf olarak

python
import fitz  # PyMuPDF
import pdfplumber
from docx import Document
import requests
import os

def extract_pdf_structure(pdf_path, image_dir="images"):
    os.makedirs(image_dir, exist_ok=True)
    doc = fitz.open(pdf_path)
    flow = []

    for page_num, page in enumerate(doc):
        blocks = page.get_text("dict")["blocks"]
        for block in blocks:
            if "image" in block:
                # Görsel çıkarımı
                pix = fitz.Pixmap(doc, block["image"])
                img_path = f"{image_dir}/page{page_num+1}_img.png"
                pix.save(img_path)
                flow.append({"type": "image", "path": img_path})
            elif "lines" in block:
                # Metin bloğu çıkarımı
                text = ""
                for line in block["lines"]:
                    for span in line["spans"]:
                        text += span["text"] + " "
                flow.append({"type": "text", "content": text.strip()})

        # Tablo çıkarımı için pdfplumber
        with pdfplumber.open(pdf_path) as plumber_doc:
            tables = plumber_doc.pages[page_num].extract_tables()
            for table in tables:
                flow.append({"type": "table", "content": table})
    
    return flow

def translate_text_blocks(flow, api_url):
    translated_flow = []
    for block in flow:
        if block["type"] == "text":
            payload = {"prompt": f"Translate this to Turkish:\n{block['content']}"}
            response = requests.post(api_url, json=payload)
            translated_text = response.json().get("translation", "")
            translated_flow.append({"type": "text", "content": translated_text})
        else:
            translated_flow.append(block)
    return translated_flow

def create_docx(flow, output_path="output.docx"):
    doc = Document()
    for block in flow:
        if block["type"] == "text":
            doc.add_paragraph(block["content"])
        elif block["type"] == "image":
            doc.add_picture(block["path"], width=None)
        elif block["type"] == "table":
            table = doc.add_table(rows=len(block["content"]),
                                  cols=len(block["content"][0]))
            for i, row in enumerate(block["content"]):
                for j, cell in enumerate(row):
                    table.cell(i, j).text = str(cell)
        doc.add_paragraph("")  # boşluk
    doc.save(output_path)

# Örnek kullanım
flow = extract_pdf_structure("sample.pdf")
translated_flow = translate_text_blocks(flow, api_url="http://localhost:8000/translate")
create_docx(translated_flow, "translated_output.docx")

🧠 Adım 1 — PDF Extraction (Genel Amaçlı Sürüm Tasarımı)

Bu modül 3 fazda işleyecek:

FAZ 1 — Layout analizi (PyMuPDF)

  • Sayfadaki tüm blokları (metin, resim) çıkarır.

  • Metin bloklarını pozisyonlarına göre sıralar (yukarıdan aşağı).

  • Görselleri alır, sıkıştırır ve diske kaydeder.

📦 Çıktı örneği:

  {"type": "image", "page": 1, "path": "images/p1_img1.jpg", "bbox": [80, 150, 500, 300]}
]

FAZ 2 — Tablo tespiti (pdfplumber + fallback OCR)

  • pdfplumber ile tablo grid yapısını çıkarmaya çalışır.

  • Eğer tablo bulunmaz ama “satır/sütun benzeri OCR metni” varsa → OCR tabanlı layout parser devreye girer.

  • Tabloları list[list[str]] formatında saklar.

FAZ 3 — OCR fallback (image-only sayfalar için)

  • Eğer sayfa tamamen resimse → pytesseract ile OCR yapılır.

  • OCR sonucu metin blokları da aynı flow içine alınır.

  • OCR’la çıkan metinler için "source": "ocr" etiketi eklenir.

FAZ 4 — Görsel Sıkıştırma

  • Görselleri .jpg olarak sıkıştıracağız (örneğin kalite = 70).

  • Bu hem DOCX’te boyutu azaltır hem de kaliteyi yeterli tutar.

  • Kütüphane: Pillow (PIL).

🧩 Örnek Kod (tam fonksiyonel prototip)

python
import fitz  # PyMuPDF
import pdfplumber
from PIL import Image
import io, os
import pytesseract

def extract_pdf(pdf_path, image_dir="images", ocr=True, image_quality=70):
    os.makedirs(image_dir, exist_ok=True)
    doc = fitz.open(pdf_path)
    full_flow = []

    for page_index, page in enumerate(doc):
        page_number = page_index + 1
        flow = []
        blocks = page.get_text("dict")["blocks"]

        # 1️⃣ Metin ve Görselleri çıkar
        for block in blocks:
            if "lines" in block:
                # Metin bloğu
                text = " ".join(span["text"] for line in block["lines"] for span in line["spans"]).strip()
                if text:
                    flow.append({
                        "type": "text",
                        "page": page_number,
                        "bbox": block["bbox"],
                        "content": text,
                        "source": "pdf"
                    })
            elif "image" in block:
                # Görsel bloğu
                xref = block["image"]
                base_image = doc.extract_image(xref)
                image_data = base_image["image"]
                img = Image.open(io.BytesIO(image_data))

                # Görsel sıkıştır
                img_path = f"{image_dir}/page{page_number}_img{len(flow)}.jpg"
                img.save(img_path, "JPEG", quality=image_quality)

                flow.append({
                    "type": "image",
                    "page": page_number,
                    "bbox": block["bbox"],
                    "path": img_path
                })

        # 2️⃣ Tablo çıkarımı (pdfplumber)
        with pdfplumber.open(pdf_path) as plumber_doc:
            page_tables = plumber_doc.pages[page_index].extract_tables()
            for table in page_tables:
                flow.append({
                    "type": "table",
                    "page": page_number,
                    "content": table
                })

        # 3️⃣ OCR fallback (eğer hiç metin bulunamadıysa)
        if ocr and not any(b["type"] == "text" for b in flow):
            pix = page.get_pixmap()
            img_path = f"{image_dir}/page{page_number}_ocr.png"
            pix.save(img_path)
            ocr_text = pytesseract.image_to_string(Image.open(img_path))
            if ocr_text.strip():
                flow.append({
                    "type": "text",
                    "page": page_number,
                    "content": ocr_text.strip(),
                    "source": "ocr"
                })

        # Sayfa bazlı sırayı pozisyona göre sırala (y0'a göre)
        flow.sort(key=lambda b: b.get("bbox", [0, 0, 0, 0])[1])
        full_flow.extend(flow)

    return full_flow

🧠 Adım 2 – LLM Entegrasyonu (Çeviri Pipeline)

Bu aşamada amacımız:

  • Extraction’dan gelen flow listesindeki "text" bloklarını alıp

⚙️ Genel Strateji

  1. Text bloklarını ayıkla

    • flow içinden sadece "type": "text" olanları seç.

    • Çok uzun metinleri LLM’in bağlam limitine göre chunk’la (örneğin 2000 karakterlik parçalara ayır).

  2. Çeviri isteği gönder

    • Her chunk’ı (veya batch’i) LLM endpoint’ine POST isteği olarak gönder.

    • Endpoint örneği (örnek vLLM API formatı):

      json
      {"prompt": "Translate the following text to Turkish:\n<text_here>"}

    • Response örneği:

      json
      {"translation": "Türkçe çeviri metni..."}

  3. Sonuçları akışa geri yerleştir

    • Orijinal sıralamayı (ve sayfa bilgisini) koru.

    • "translated_content" alanı ekle.

🧩 Kod – translate_flow() Fonksiyonu

Aşağıdaki örnek kod:

  • Chunking yapar

  • vLLM endpoint’ine istek atar

  • Yanıtları aynı sırada flow yapısına geri ekler

  • Gerektiğinde retry mekanizması uygular

python
import requests
import time

def chunk_text(text, max_length=2000):
    """Uzun metinleri LLM bağlamına uygun parçalara böler."""
    chunks = []
    current = ""
    for sentence in text.split(". "):
        if len(current) + len(sentence) < max_length:
            current += sentence + ". "
        else:
            chunks.append(current.strip())
            current = sentence + ". "
    if current:
        chunks.append(current.strip())
    return chunks

def call_translation_api(api_url, text, retries=3, sleep=2):
    """vLLM endpoint’ine çeviri isteği gönderir."""
    payload = {"prompt": f"Translate the following text to Turkish:\n{text}"}
    for attempt in range(retries):
        try:
            resp = requests.post(api_url, json=payload, timeout=60)
            resp.raise_for_status()
            data = resp.json()
            if "translation" in data:
                return data["translation"]
            elif "output" in data:
                # Bazı modeller output key’inde döner
                return data["output"]
        except Exception as e:
            print(f"⚠️ API hatası ({attempt+1}/{retries}): {e}")
            time.sleep(sleep)
    return text  # fallback: orijinal metin

def translate_flow(flow, api_url, batch_size=3, max_chunk_length=2000):
    """Flow içindeki metin bloklarını çevirir."""
    translated_flow = []
    text_blocks = [b for b in flow if b["type"] == "text"]

    for i, block in enumerate(text_blocks):
        text = block["content"]
        chunks = chunk_text(text, max_length=max_chunk_length)
        translated_chunks = []
        print(f"🔤 Çeviri başlıyor (blok {i+1}/{len(text_blocks)}, {len(chunks)} parça)...")

        for ch in chunks:
            translated = call_translation_api(api_url, ch)
            translated_chunks.append(translated)

        block["translated_content"] = " ".join(translated_chunks)
        translated_flow.append(block)

    # Metin olmayan blokları da ekle
    for b in flow:
        if b["type"] != "text":
            translated_flow.append(b)

    # Sıralamayı koru
    translated_flow.sort(key=lambda b: (b["page"], b.get("bbox", [0, 0, 0, 0])[1]))
    return translated_flow

python
from docx import Document
from docx.shared import Inches, Pt
from docx.enum.text import WD_ALIGN_PARAGRAPH
from PIL import Image

def create_docx_from_flow(flow, output_path="translated_output.docx", bilingual=False, image_max_width=5.5):
    """
    flow: translated_flow listesi
    output_path: kaydedilecek docx dosya adı
    bilingual: True => orijinal + çeviri, False => sadece çeviri
    image_max_width: görselin maksimum genişliği (inch)
    """
    doc = Document()
    
    # Başlık ekle
    doc.add_heading("Translated Document", level=1)
    doc.add_paragraph("")  # boş satır
    
    for block in flow:
        btype = block["type"]

        # 1️⃣ Metin blokları
        if btype == "text":
            original = block.get("content", "").strip()
            translated = block.get("translated_content", "").strip()

            if bilingual:
                # Orijinal metin
                p1 = doc.add_paragraph(original)
                p1.style = doc.styles["Normal"]
                p1.runs[0].font.size = Pt(10)
                p1.alignment = WD_ALIGN_PARAGRAPH.LEFT

                # Çeviri metni
                p2 = doc.add_paragraph(translated)
                p2.style = doc.styles["Normal"]
                p2.runs[0].font.bold = True
                p2.runs[0].font.size = Pt(11)
                p2.alignment = WD_ALIGN_PARAGRAPH.LEFT

                doc.add_paragraph("")  # boşluk
            else:
                p = doc.add_paragraph(translated or original)
                p.style = doc.styles["Normal"]
                p.alignment = WD_ALIGN_PARAGRAPH.LEFT
                doc.add_paragraph("")  # boşluk

        # 2️⃣ Görseller
        elif btype == "image":
            try:
                img = Image.open(block["path"])
                width_inch = min(image_max_width, img.width / img.info.get("dpi", (72,))[0])
                doc.add_picture(block["path"], width=Inches(width_inch))
                doc.add_paragraph("")  # boşluk
            except Exception as e:
                print(f"⚠️ Görsel eklenemedi: {e}")

        # 3️⃣ Tablolar
        elif btype == "table":
            table_data = block.get("content", [])
            if not table_data:
                continue
            table = doc.add_table(rows=len(table_data), cols=len(table_data[0]))
            table.style = "Table Grid"
            for i, row in enumerate(table_data):
                for j, cell in enumerate(row):
                    table.cell(i, j).text = str(cell)
            doc.add_paragraph("")  # tablo sonrası boşluk

    doc.save(output_path)
    print(f"✅ DOCX dosyası oluşturuldu: {output_path}")

import fitz
import pdfplumber
from PIL import Image
import pytesseract
import io, os

def extract_pdf(pdf_path, image_dir, ocr=True, image_quality=70):
    os.makedirs(image_dir, exist_ok=True)
    doc = fitz.open(pdf_path)
    full_flow = []

    for page_index, page in enumerate(doc):
        page_number = page_index + 1
        flow = []
        blocks = page.get_text("dict")["blocks"]

        for block in blocks:
            if "lines" in block:
                text = " ".join(span["text"] for line in block["lines"] for span in line["spans"]).strip()
                if text:
                    flow.append({
                        "type": "text",
                        "page": page_number,
                        "bbox": block["bbox"],
                        "content": text,
                        "source": "pdf"
                    })
            elif "image" in block:
                xref = block["image"]
                base_image = doc.extract_image(xref)
                image_data = base_image["image"]
                img = Image.open(io.BytesIO(image_data))
                img_path = f"{image_dir}/page{page_number}_img{len(flow)}.jpg"
                img.save(img_path, "JPEG", quality=image_quality)
                flow.append({
                    "type": "image",
                    "page": page_number,
                    "bbox": block["bbox"],
                    "path": img_path
                })

        # pdfplumber tablo çıkarımı
        with pdfplumber.open(pdf_path) as plumber_doc:
            tables = plumber_doc.pages[page_index].extract_tables()
            for table in tables:
                flow.append({
                    "type": "table",
                    "page": page_number,
                    "content": table
                })

        # OCR fallback
        if ocr and not any(b["type"] == "text" for b in flow):
            pix = page.get_pixmap()
            img_path = f"{image_dir}/page{page_number}_ocr.png"
            pix.save(img_path)
            ocr_text = pytesseract.image_to_string(Image.open(img_path))
            if ocr_text.strip():
                flow.append({
                    "type": "text",
                    "page": page_number,
                    "content": ocr_text.strip(),
                    "source": "ocr"
                })

        flow.sort(key=lambda b: b.get("bbox", [0, 0, 0, 0])[1])
        full_flow.extend(flow)

    return full_flow

🧠 app/pipeline/translator.py

python
import requests
import time

def chunk_text(text, max_length=2000):
    chunks, current = [], ""
    for sentence in text.split(". "):
        if len(current) + len(sentence) < max_length:
            current += sentence + ". "
        else:
            chunks.append(current.strip())
            current = sentence + ". "
    if current:
        chunks.append(current.strip())
    return chunks

def call_translation_api(api_url, text, retries=3, sleep=2):
    payload = {"prompt": f"Translate the following text to Turkish:\n{text}"}
    for attempt in range(retries):
        try:
            resp = requests.post(api_url, json=payload, timeout=60)
            resp.raise_for_status()
            data = resp.json()
            return data.get("translation") or data.get("output") or text
        except Exception as e:
            print(f"⚠️ API hatası ({attempt+1}/{retries}): {e}")
            time.sleep(sleep)
    return text

def translate_flow(flow, api_url):
    translated_flow = []
    text_blocks = [b for b in flow if b["type"] == "text"]

    for i, block in enumerate(text_blocks):
        text = block["content"]
        chunks = chunk_text(text)
        translated_chunks = []
        print(f"🔤 Blok {i+1}/{len(text_blocks)} çeviriliyor...")

        for ch in chunks:
            translated = call_translation_api(api_url, ch)
            translated_chunks.append(translated)

        block["translated_content"] = " ".join(translated_chunks)
        translated_flow.append(block)

    # Metin olmayanları da geri ekle
    for b in flow:
        if b["type"] != "text":
            translated_flow.append(b)

    translated_flow.sort(key=lambda b: (b["page"], b.get("bbox", [0, 0, 0, 0])[1]))
    return translated_flow

from docx import Document
from docx.shared import Inches, Pt
from docx.enum.text import WD_ALIGN_PARAGRAPH
from PIL import Image

def create_docx_from_flow(flow, output_path, bilingual=False, image_max_width=5.5):
    doc = Document()
    doc.add_heading("Translated Document", level=1)
    doc.add_paragraph("")

    for block in flow:
        if block["type"] == "text":
            original = block.get("content", "").strip()
            translated = block.get("translated_content", "").strip()

            if bilingual:
                p1 = doc.add_paragraph(original)
                p1.runs[0].font.size = Pt(10)
                p2 = doc.add_paragraph(translated)
                p2.runs[0].font.bold = True
                p2.runs[0].font.size = Pt(11)
                doc.add_paragraph("")
            else:
                p = doc.add_paragraph(translated or original)
                p.alignment = WD_ALIGN_PARAGRAPH.LEFT
                doc.add_paragraph("")

        elif block["type"] == "image":
            try:
                img = Image.open(block["path"])
                doc.add_picture(block["path"], width=Inches(image_max_width))
                doc.add_paragraph("")
            except Exception as e:
                print(f"⚠️ Görsel eklenemedi: {e}")

        elif block["type"] == "table":
            table_data = block.get("content", [])
            if not table_data:
                continue
            table = doc.add_table(rows=len(table_data), cols=len(table_data[0]))
            table.style = "Table Grid"
            for i, row in enumerate(table_data):
                for j, cell in enumerate(row):
                    table.cell(i, j).text = str(cell)
            doc.add_paragraph("")

    doc.save(output_path)
    return output_path

⚡ app/main.py

python
from fastapi import FastAPI, UploadFile, File
from fastapi.responses import FileResponse, JSONResponse
import os, uuid

from app.config import UPLOAD_DIR, OUTPUT_DIR, IMAGE_DIR, LLM_API_URL
from app.pipeline.extractor import extract_pdf
from app.pipeline.translator import translate_flow
from app.pipeline.docx_creator import create_docx_from_flow

app = FastAPI(title="PDF Translator API", version="1.0.0")

@app.post("/translate")
async def translate_pdf(file: UploadFile = File(...), bilingual: bool = False):
    try:
        # 1️⃣ PDF kaydet
        file_id = str(uuid.uuid4())
        pdf_path = os.path.join(UPLOAD_DIR, f"{file_id}.pdf")
        with open(pdf_path, "wb") as f:
            f.write(await file.read())

        # 2️⃣ Extraction
        flow = extract_pdf(pdf_path, image_dir=IMAGE_DIR)

        # 3️⃣ Translation
        translated_flow = translate_flow(flow, api_url=LLM_API_URL)

        # 4️⃣ DOCX oluştur
        output_path = os.path.join(OUTPUT_DIR, f"{file_id}.docx")
        create_docx_from_flow(translated_flow, output_path, bilingual=bilingual)

        return FileResponse(
            output_path,
            media_type="application/vnd.openxmlformats-officedocument.wordprocessingml.document",
            filename=f"translated_{file.filename}.docx"
        )

    except Exception as e:
        return JSONResponse({"error": str(e)}, status_code=500)
